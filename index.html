/** * ネム勤怠 × Googleカレンダー 書き込み同期スクリプト * ================================================ * 【設定手順】 * 1. https://script.google.com にアクセス * 2. 新しいプロジェクト → このコードをすべて貼り付け * 3. 「デプロイ」→「新しいデプロイ」→ 種類: ウェブアプリ * 4. 実行ユーザー: 自分、アクセス: 全員（匿名含む）→「デプロイ」 * 5. 表示されたURLをネム勤怠の「管理設定 → GAS Webhook URL」に貼り付け */ function doPost(e) { try { const payload = JSON.parse(e.postData.contents); if (payload.action === 'syncShifts') { return syncShifts(payload); } return jsonResponse({ status: 'error', message: '不明なアクション' }); } catch (err) { return jsonResponse({ status: 'error', message: err.message }); } } function syncShifts(payload) { const month = payload.month; // "2026-03" const shifts = payload.shifts || []; let added = 0, updated = 0, errors = 0; for (const shift of shifts) { if (!shift.calId || !shift.date || !shift.start || !shift.end) continue; try { const cal = CalendarApp.getCalendarById(shift.calId); if (!cal) { errors++; continue; } const startDt = new Date(`${shift.date}T${shift.start}:00`); const endDt = new Date(`${shift.date}T${shift.end}:00`); const title = `${shift.empName || shift.empId} シフト${shift.note ? ' ' + shift.note : ''}`; // 既存イベントがあれば更新（gcalId で照合） if (shift.gcalId) { try { const existing = cal.getEventById(shift.gcalId); if (existing) { existing.setTitle(title); existing.setTime(startDt, endDt); updated++; continue; } } catch (e) { /* イベントが見つからない場合は新規作成へ */ } } // 同じ日の同じタイトルの重複チェック const dayStart = new Date(`${shift.date}T00:00:00`); const dayEnd = new Date(`${shift.date}T23:59:59`); const existing = cal.getEvents(dayStart, dayEnd, { search: title }); if (existing.length > 0) { // 既存を更新 existing[0].setTime(startDt, endDt); updated++; } else { // 新規追加 cal.createEvent(title, startDt, endDt); added++; } } catch (err) { errors++; console.error('shift error:', err.message, shift); } } return jsonResponse({ status: 'ok', month: month, added: added, updated: updated, errors: errors, message: `${added}件追加、${updated}件更新、${errors}件エラー` }); } function jsonResponse(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } // テスト用（スクリプトエディタから直接実行して動作確認） function testSync() { const testPayload = { action: 'syncShifts', month: '2026-03', shifts: [ { id: 'test_001', empId: 'emp001', calId: '2776d3b89ac88c89f86bb9890ac836de358ba5459f0c1c2e133f7694f804f2bb@group.calendar.google.com', date: '2026-03-15', start: '09:00', end: '17:00', note: 'テスト', empName: '田中太郎' } ] }; const result = syncShifts(testPayload); console.log(result.getContent()); }
